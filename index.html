The Problem with AI-Generated Code

📦

Wrong Libraries

AI uses libraries that don't exist in your project. It suggests axios when you use fetch, creating unnecessary dependencies and inconsistencies.

🔤

Naming Chaos

AI follows different naming conventions. It writes snake_case when your repo uses camelCase, breaking your codebase standards and readability.

🔄

Reinventing the Wheel

AI reimplements utilities that already exist in your codebase. It creates duplicate functions instead of using your existing, tested helpers.

Before & After

❌ Before: AI-Generated Code
const axios = require('axios');  // ❌ Wrong library

async function create_new_user(user_data) {  // ❌ snake_case
  // ❌ No try/catch
  
  const email_regex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;  // ❌ Duplicate
  if (!email_regex.test(user_data.email)) {
    return { success: false, error: 'Invalid email' };
  }
  
  const response = await axios.post('https://api.example.com/users', user_data);
  return { success: true, data: response.data };
}
✅ After: DebugLens Fixed
const { isValidEmail } = require('./utils');  // ✅ Existing utility
const { handleError, handleResponse } = require('./handlers');

async function createNewUser(userData) {  // ✅ camelCase
  try {  // ✅ Error handling
    if (!userData.email || !isValidEmail(userData.email)) {
      throw new Error('Valid email required');
    }
    
    const response = await fetch('https://api.example.com/users', {  // ✅ fetch
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(userData)
    });
    
    const data = await response.json();
    return handleResponse(data, response.status);
  } catch (error) {
    return handleError(error, 'createNewUser');
  }
}
